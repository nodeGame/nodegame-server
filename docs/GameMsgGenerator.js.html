GameMsgGenerator.prototype.createACK = function (gm, to, reliable) {
  
  var rel = reliable || 0;
  
  var newgm = new GameMsg({
              session: this.session, 
              state: this.state,
              action: GameMsg.actions.SAY,
              target: GameMsg.targets.ACK,
              from: this.sender,
              to: to,
              text: 'Msg ' + gm.id + ' correctly received',
              data: gm.id,
              priority: null,
              reliable: rel
  });
  
  if (gm.forward) {
    newgm.forward = 1;
  }
  
  return newgm;
};
